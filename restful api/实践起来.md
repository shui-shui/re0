###### 状态码

状态码	含义

200	成功

201	创建成功（POST）

400	请求错误

401	未授权

404	找不到，资源没找到

500	服务器错误



##### 题目：

**requests.post(url, data=data) 不加 JSON**

后端期望 JSON 格式

你只传普通字符串（或者 form-data），服务器不知道这是 JSON

会返回 400 错误 或解析失败

**Authorization: Bearer xxx**

Authorization = 身份凭证

后端通过 token 识别你是谁、是否有权限访问接口

import requests

BASE_URL = "https://jsonplaceholder.typicode.com/users"
TOKEN = "my_fake_token"

# headers
HEADERS = {
    "Authorization": f"Bearer {TOKEN}",
    "Content-Type": "application/json"
}

def get_user(user_id):
    """获取用户信息"""
    url = f"{BASE_URL}/{user_id}"
    response = requests.get(url, headers=HEADERS)

    if response.status_code == 200:
        data = response.json()
        print(f"用户信息: {data}")
    else:
        print(f"接口调用失败，状态码 {response.status_code}")

def create_user(name, email):
    """创建新用户"""
    payload = {
        "name": name,
        "email": email
    }
    response = requests.post(BASE_URL, json=payload, headers=HEADERS)

    if response.status_code in [200, 201]:
        data = response.json()
        print(f"新用户创建成功: {data}")
    else:
        print(f"接口调用失败，状态码 {response.status_code}")

import requests

BASE_URL = "https://jsonplaceholder.typicode.com/users"
TOKEN = "my_fake_token"

HEADERS = {
    "Authorization": f"Bearer {TOKEN}",
    "Content-Type": "application/json"
}

def get_user():
    user_id = input("请输入用户 ID: ")
    url = f"{BASE_URL}/{user_id}"
    response = requests.get(url, headers=HEADERS)

    if response.status_code == 200:
        print("✅ 用户信息:", response.json())
    elif response.status_code == 404:
        print("❌ 用户不存在")
    else:
        print(f"❌ 请求失败，状态码 {response.status_code}")

def create_user():
    name = input("请输入用户名: ")
    email = input("请输入用户邮箱: ")

    payload = {
        "name": name,
        "email": email
    }

    response = requests.post(BASE_URL, json=payload, headers=HEADERS)

    if response.status_code in [200, 201]:
        print("✅ 新用户创建成功:", response.json())
    else:
        print(f"❌ 创建失败，状态码 {response.status_code}")

def main():
    while True:
        print("\n=== 用户管理系统 ===")
        print("1. 查询用户")
        print("2. 创建用户")
        print("3. 退出")

        choice = input("请选择操作 (1/2/3): ")

        if choice == "1":
            get_user()
        elif choice == "2":
            create_user()
        elif choice == "3":
            print("退出系统")
            break
        else:
            print("无效选项，请重新选择")

if __name__ == "__main__":
    main()
